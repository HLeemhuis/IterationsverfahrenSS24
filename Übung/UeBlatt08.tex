\item $A^H J = J^H A, \quad J\in \mathbb{C}^{n\times n}, \quad J = J^H$ (man nennt $A$ dann auch \textit{J-hermitesch}
\item $A^T J = J^T A, \quad J \in \mathbb{C}^{n\times n}$ (man nennt $A$ dann auch \textit{J-symmetrisch})
\end{itemize}
