%% Algorithms %%
%\usepackage{myalgorithmic}
\usepackage{etoolbox}
\usepackage[vlined,ruled,linesnumbered]{algorithm2e}
  % add tightalgo environment since the algorithm environment spans the whole width of the page
  % this is from http://tex.stackexchange.com/questions/23296/setting-caption-rule-width-in-algorithm2e-algs
  \newlength{\tightalgowidth}
  \newlength{\tightalgoremainder}
  \newenvironment{tightalgo}[2][]
  { \setlength{\tightalgowidth}{#2}
    \setlength{\tightalgoremainder}{\linewidth-\tightalgowidth}
    \begin{algorithm}[#1] }
  { \end{algorithm} }

  \makeatletter
  \patchcmd{\@algocf@start}{%
    \begin{lrbox}{\algocf@algobox}%
  }{%
    \rule{0.5\tightalgoremainder}{\z@}%
    \begin{lrbox}{\algocf@algobox}%
    \begin{minipage}{\tightalgowidth}%
  }{}{}
  \patchcmd{\@algocf@finish}{%
    \end{lrbox}%
  }{%
    \end{minipage}%
    \end{lrbox}%
  }{}{}
  \makeatother
